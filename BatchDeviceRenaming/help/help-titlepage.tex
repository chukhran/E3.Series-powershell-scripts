\startcomponent help-titlepage

\environment help-environment

\setvariables
  [document]
  [title=Batch Device,
   subtitle=Renaming,
   author={\E}]

\startpagemakeup

    \startMPcode

        StartPage ;

            fill Page enlarged 5mm withcolor \MPcolor{titlepagecolor} ;

            draw anchored.lrt(
                image(
                    draw textext("\getvariable{document}{title}")
                    xsized(.50PaperWidth)
                    withcolor white
                ),
                (lrcorner Page) shifted (-PaperWidth/20, PaperWidth/3.8)
            ) ;

            draw anchored.lrt(image(draw textext("\getvariable{document}{subtitle}") xsized(.750PaperWidth)            withcolor white),(lrcorner Page) shifted (-PaperWidth/20, PaperWidth/10)) ;
            draw anchored.urt(image(draw textext("\getvariable{document}{author}")   xsized(.375PaperWidth) rotated 90 withcolor white),(urcorner Page) shifted (-PaperWidth/20,-PaperWidth/20)) ;

            setbounds currentpicture to Page ;

        StopPage ;

    \stopMPcode

\stoppagemakeup

\stopcomponent
